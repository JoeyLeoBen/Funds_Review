WITH x_view AS(
	SELECT *
	FROM (
		SELECT fund_identifier, fund_category, fund_company_id
		FROM analytics.raw_canada_life_fund_categories
		UNION 
		SELECT fund_identifier, fund_category, fund_company_id
		FROM analytics.raw_manulife_fund_categories
		UNION
		SELECT fund_identifier, fund_category, fund_company_id
		FROM analytics.raw_sun_life_fund_categories
	) AS bmv
	LEFT JOIN(
		SELECT fund_identifier, sub_peer_group, 'PEER GROUP' AS peer_group_marker
		FROM analytics.raw_canada_life_benchmarks
		UNION
		SELECT fund_identifier, sub_peer_group, 'PEER GROUP' AS peer_group_marker
		FROM analytics.raw_manulife_benchmarks
	) AS fc
	ON bmv.fund_identifier = fc.fund_identifier
	ORDER BY bmv.fund_identifier
)

SELECT *
FROM x_view
